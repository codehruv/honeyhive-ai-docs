---
title: "Create Task"
api: "POST https://honeyhive.ai:5000/tasks"
description: "This endpoint creates a new task."
---

### Body

<ParamField body="name" type="string">
  Required parameter to specify the name of the task.
</ParamField>

<ParamField body="description" type="string">
  Optional parameter to specify the description of the task.
</ParamField>

<ParamField body="type" type="string">
  Optional parameter to specify the type of the task. Possible values are 'Text Generation', 'Code Generation', 'Classification', 'Summarization' and 'Other'.
</ParamField>

<ParamField body="fine_tuned_models" type="array">
  Optional parameter to specify the list of fine-tuned models for the task.
</ParamField>

<ParamField body="prompts" type="array">
  Optional parameter to specify the list of prompts for the task.
</ParamField>

<ParamField body="datasets" type="array">
  Optional parameter to specify the list of datasets for the task.
</ParamField>

<ParamField body="metrics" type="array">
  Optional parameter to specify the list of metrics for the task.
</ParamField>

### Response

<ResponseField name="data" type="object">

The contents of the created task

<Expandable title="Toggle object">

<ResponseField name="name" type="string">
  This is the name of the task.
</ResponseField>

<ResponseField name="type" type="string">
  This is the type of the task. Possible values are 'Text Generation', 'Code Generation', 'Classification', 'Summarization' and 'Other'.
</ResponseField>

<ResponseField name="prompts" type="array">
  This is the list of prompts saved for the task.
</ResponseField>

<ResponseField name="fine_tuned_models" type="array">
  This is the list of fine-tuned models that have been trained for the task.
</ResponseField>

<ResponseField name="datasets" type="array">
  This is the list of datasets that have been uploaded for the task.
</ResponseField>

<ResponseField name="metrics" type="array">
  This is the list of metrics that have been setup for the task.
</ResponseField>

<ResponseField name="created_at" type="string">
  This is the date and time when the task was created.
</ResponseField>

<ResponseField name="updated_at" type="string">
  This is the date and time when the task was last updated.
</ResponseField>

<ResponseField name="id" type="string">
  This is the unique identifier for the task.
</ResponseField>

</Expandable>

</ResponseField>
