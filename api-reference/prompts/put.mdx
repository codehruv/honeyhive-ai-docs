---
title: "Change Prompt"
api: "PUT https://honeyhive.ai:5000/prompts/:id"
description: "This endpoint changes a prompt for a task."
---

### Body

<ParamField body="name" type="string">
  The name of the prompt to be changed.
</ParamField>

<ParamField body="version" type="string">
  The version of the prompt to be changed.
</ParamField>

<ParamField body="text" type="string">
  The text of the prompt to be changed.
</ParamField>

<ParamField body="input_variables" type="array">
  The list of input variables for the prompt to be changed.
</ParamField>

<ParamField body="model" type="string">
  The model to be used for the prompt to be changed.
</ParamField>

<ParamField body="provider" type="string">
  The provider to be used for the prompt to be changed.
</ParamField>

<ParamField body="hyperparameters" type="object">
  The hyperparameters to be used for the prompt to be changed.
</ParamField>

<ParamField body="is_deployed" type="boolean">
  Whether the prompt is deployed or not.
</ParamField>

<ParamField body="few_shot_examples" type="array">
  The list of few-shot examples for the prompt to be changed.
</ParamField>

### Response

<ResponseField name="data" type="object">

The contents of the updated prompt

  This is the hyperparameters used for the prompt.
</ResponseField>

<ResponseField name="is_deployed" type="boolean">

</ResponseField>
